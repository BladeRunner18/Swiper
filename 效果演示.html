                    //如果需要和swiper插件同时使用  将第三行代码global.Swiper = factory();中的Swiper改成任意别的名字 
                    // 就不会与swiper插件冲突
                        //     slidesPerView : 1,    每一屏的数量      同swiper  
                        //     spaceBetween : 0,     边距             同swiper  
                        //     loop:false,           循环             同swiper
                        //     autoplay:null,        自动轮播         同swiper  
                        //     speed:500,            切换速度         同swiper
                        //     navigation:null,      左右切换按钮      同swiper
                        //     pagination:null,      分页的点          同swiper
                        //     breakPoints:{},       响应式参数        同swiper
                        //     thumb:null,           缩略图            同swiper
                        //     drag:true,            开启或关闭拖拽切换        
                        //     touch:true,           开启活关闭滑动切换         
                        //     responsive:true,      开启或关闭响应式          
                        //     useData:false,        布局的时候使用         
                        //     play:true,            布局的时候使用
